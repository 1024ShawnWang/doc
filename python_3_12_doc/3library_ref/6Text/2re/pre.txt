import re

#1.python会调用解释器对字符串类型的对象进行加工，默认情况下，主要是一些特殊字符，用于控制
#比如 '\n'，并不会在屏幕上打印出 \n ，而是换行
print('\n')

#2.强制python解释器不对字符串进行解释，需要在字符串前面增加r这个字符
print( r'\n' )


#3.类比 在python中  写入的字符 同 最终的打印效果， 正则表达式对于 输出字符和匹配结果 有一样的设计
#比如，正则表达式\\，才匹配一个\


#4.因此，在python中，想要写一个字符串匹配\ ，不刻意指定r 的情况下，得这样写 \\\\
